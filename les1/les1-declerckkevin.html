Voorbeeld met rekenmachine

function Rekenmachine() {
	this.waarde1 = 0;
	this.waarde2 = 0;

	this.Optellen = function(waarde1, waarde2) {
						return waarde1 + waarde2;
					}

	this.Aftrekken = function(waarde1, waarde2) {
						return waarde1 - waarde2;
					}

	this.Vermenigvuldigen = function(waarde1, waarde2) {
						return waarde1 * waarde2;
					}

	this.Deel = function(waarde1, waarde2) {
						return waarde1 / waarde2;
					}
	
var rekenen = new Rekenmachine();
								
var uitkomst = rekenen.Optellen();
document.write("uitkomst optellen : " + uitkomst + '
');
							
uitkomst = rekenen.Aftrekken();
document.write("uitkomst aftrekken: " + uitkomst + '
');
								
uitkomst = rekenen.Vermenigvuldigen();
document.write("uitkomst vermenigvuldigen: " + uitkomst + '
');
								
uitkomst = rekenen.Deel();
document.write("uitkomst : delen" + uitkomst + '
');

}
